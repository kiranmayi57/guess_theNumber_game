#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int guess, number;
    srand(time(0));
    number = rand() % 10 + 1;

    printf("Guess a number (1-10): ");
    scanf("%d", &guess);

    if(guess == number)
        printf("Correct Guess!\n");
    else
        printf("Wrong! Number was %d\n", number);

    return 0;
}
